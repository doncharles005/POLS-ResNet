        self.fc2 = nn.Linear(hidden_size, hidden_size)

        self.fc3 = nn.Linear(hidden_size, output_size)

        out = self.fc1(x)
        out = self.relu(out)
        out = self.fc2(out)
        out = self.relu2(out)
        out = self.fc3(out)

# Adjusting based on the actual column names

independent_vars = ['ODA', 'debt']  # Independent variables

# Standardizing the independent variables
scaler = StandardScaler()
df[independent_vars] = scaler.fit_transform(df[independent_vars])

# Dependent variable (BoT)
y = df[dependent_var]

# Independent variables (ODA and debt)
X = df[independent_vars]

# Adding constant to the model for the intercept
X = sm.add_constant(X)

# Step 1: Initial Linear Estimation (POLS)
model = sm.OLS(y, X).fit()

# Displaying POLS results
print(model.summary())

# Step 2: Calculate residuals

# Step 3: Prepare data for RNN
# Reshape input for RNN
X_rnn = X.values.reshape(X.shape[0], X.shape[1], 1)  # Reshaping for RNN

# Define the RNN model
class RNN(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(RNN, self).__init__()
        self.rnn = nn.RNN(input_size, hidden_size, batch_first=True)
        self.fc = nn.Linear(hidden_size, output_size)

    def forward(self, x):
        out, _ = self.rnn(x)
        out = self.fc(out[:, -1, :])  # Get the last time step's output
        return out

# Model parameters

hidden_size = 10  # Number of RNN hidden units
output_size = 1  # Predicting one output (the residual)

# Instantiate the RNN model
rnn_model = RNN(input_size, hidden_size, output_size)

# Loss and optimizer
criterion = nn.MSELoss()
optimizer = optim.Adam(rnn_model.parameters(), lr=0.001)

# Step 4: Train the RNN
num_epochs = 100  # Adjust as needed
for epoch in range(num_epochs):
    rnn_model.train()
    
    # Convert data to torch tensors
    inputs = torch.tensor(X_rnn, dtype=torch.float32)
    targets = torch.tensor(residuals.values, dtype=torch.float32).view(-1, 1)  # Reshape targets

    # Forward pass
    outputs = rnn_model(inputs)
    loss = criterion(outputs, targets)
    
    # Backward pass and optimization
    optimizer.zero_grad()
    loss.backward()
    optimizer.step()
    
    if (epoch + 1) % 10 == 0:
        print(f'Epoch [{epoch + 1}/{num_epochs}], Loss: {loss.item():.4f}')

# Step 5: Combine POLS and RNN Predictions
with torch.no_grad():
    rnn_model.eval()
    rnn_outputs = rnn_model(inputs).numpy()
    
# Final predictions

# Display final predictions
print("Final predictions combining POLS and RNN:")
print(final_predictions)

y = df['BoT']

class ANNModel(nn.Module):
    def __init__(self, input_size, hidden_size, output_size):
        super(ANNModel, self).__init__()
        self.fc1 = nn.Linear(input_size, hidden_size)
        self.relu = nn.ReLU()
        self.fc2 = nn.Linear(hidden_size, hidden_size)
        self.relu2 = nn.ReLU()
        self.fc3 = nn.Linear(hidden_size, output_size)
    
    def forward(self, x):
        out = self.fc1(x)
        out = self.relu(out)
        out = self.fc2(out)
        out = self.relu2(out)
        out = self.fc3(out)
        return out
